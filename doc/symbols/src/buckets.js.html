 20 	var common = {};
 21 	common.DEFAULT_COMPARE = function(a, b) {
 22 
 23 		if (a < b) {
 24 			return -1;
 25 		}
 26 		else if (a === b) {
 27 			return 0;
 28 		}
 29 		else {
 30 			return 1;
 31 		}
 32 	};
 33 
 34 	common.DEFAULT_EQUALS = function(a, b) {
 35 		return a === b;
 36 	};
 37 
 38 	common.DEFAULT_TO_STRING = function(obj) {
 39 		if (Object.prototype.toString.call(obj) === '[object String]') {
 40 			return obj;
 41 		}
 42 		else {
 43 			return obj.toString();
 44 		}
 45 	};
 46 
 47 	common.createReverseCompareFunction = function(compareFunction) {
 48 		if (compareFunction === null || compareFunction === undefined) {
 49 
 50 			return function(a, b) {
 51 				if (a < b) {
 52 					return -1;
 53 				}
 54 				else if (a === b) {
 55 					return 0;
 56 				}
 57 				else {
 58 					return 1;
 59 				}
 60 			};
 61 		}
 62 		else {
 63 			return function(d, v) {
 64 				return compareFunction(d, v) * -1;
 65 			};
 66 		}
 67 	};

 73 	/**
 74 	 * Creates an empty Linked List.
 75 	 * @class A linked list is a data structure consisting of a group of nodes
 76 	 * which together represent a sequence. If the elements inside the list are
 77 	 * not comparable with the === operator a custom equals function should be
 78 	 * provided to perform searches, the function must receive two arguments and
 79 	 * return true if they are equal, false otherwise. Example:
 80 	 * 
 81 	 * <pre>
 82 	 * var petsAreEqualByName = function(pet1, pet2) {
 83 	 * 	return pet1.name === pet2.name;
 84 	 * }
 85 	 * 
 86 	 * var petList = new buckets.LinkedList(petsAreEqualByName);
 87 	 * var numberList = new buckets.LinkedList();
 88 	 * 
 89 	 * </pre>
 90 	 * 
 91 	 * @constructor
 92 	 * @this {buckets.LinkedList}
 93 	 * @param {function(Object,Object):boolean=} equalsFunction Optional
 94 	 * function used to check if two elements are equal.
 95 	 */
 96 	buckets.LinkedList = function(equalsFunction) {
 97 
 98 		/**
 99 		 * First node in the list
100 		 * @type {Object}
101 		 * @private
102 		 */
103 		this.firstNode = null;
104 
105 		/**
106 		 * Last node in the list
107 		 * @type {Object}
108 		 * @private
109 		 */
110 		this.lastNode = null;
111 
112 		/**
113 		 * Number of elements in the list
114 		 * @type {number}
115 		 * @private
116 		 */
117 		this.nElements = 0;
118 
119 		/**
120 		 * Function used to check if two elements are equal.
121 		 * @type {function(Object,Object):boolean}
122 		 * @private
123 		 */
124 		this.equalsF = equalsFunction || common.DEFAULT_EQUALS;
125 	};
126 
127 	/**
128 	 * Adds the given object to the end of this list.
129 	 * @this {buckets.LinkedList}
130 	 * @param {Object} element Object to be added.
131 	 * @returns {boolean} always returns true.
132 	 */
133 	buckets.LinkedList.prototype.add = function add(element) {
134 		return this.addElementAtIndex(element, this.nElements);
135 	};
136 
137 	/**
138 	 * Adds the given object at the specified position in this list.
139 	 * @this {buckets.LinkedList}
140 	 * @param {Object} elem object to be added.
141 	 * @param {number} index index to add the element.
142 	 * @returns {boolean} true if the element was added or false if index < 0 ||
143 	 * index > this.nElements.
144 	 */
145 	buckets.LinkedList.prototype.addElementAtIndex = function(elem, index) {
146 
147 		if (index < 0 || index > this.nElements) {
148 			return false;
149 		}
150 		var newNode = {
151 			element : elem,
152 			next : null
153 		};
154 		if (this.nElements === 0) {
155 			// First node in the list.
156 			this.firstNode = newNode;
157 			this.lastNode = newNode;
158 		}
159 		else if (index === this.nElements) {
160 			// Change this.lastNode node.
161 			this.lastNode.next = newNode;
162 			this.lastNode = newNode;
163 		}
164 		else if (index === 0) {
165 			// Change this.firstNode node.
166 			newNode.next = this.firstNode;
167 			this.firstNode = newNode;
168 		}
169 		else {
170 			var currentNode = this.firstNode;
171 			var i = 0;
172 			for (; i < (index - 1); i++) {
173 				currentNode = currentNode.next;
174 			}
175 			newNode.next = currentNode.next;
176 			currentNode.next = newNode;
177 		}
178 		this.nElements++;
179 		return true;
180 	};
181 
182 	/**
183 	 * Returns the first element in this list.
184 	 * @this {buckets.LinkedList}
185 	 * @return {*} the first element of the list or undefined if the list is
186 	 * empty.
187 	 */
188 	buckets.LinkedList.prototype.first = function() {
189 
190 		if (this.firstNode !== null) {
191 			return this.firstNode.element;
192 		}
193 		return undefined;
194 	};
195 
196 	/**
197 	 * Returns the last element in this list.
198 	 * @this {buckets.LinkedList}
199 	 * @return {*} the last element of the list or undefined if the list is
200 	 * empty.
201 	 */
202 	buckets.LinkedList.prototype.last = function() {
203 
204 		if (this.lastNode !== null) {
205 			return this.lastNode.element;
206 		}
207 		return undefined;
208 	};
209 
210 	/**
211 	 * Returns the element at the specified position in this list.
212 	 * @this {buckets.LinkedList}
213 	 * @param {number} index desired index.
214 	 * @return {*} the element at the given index or undefined if index < 0 ||
215 	 * index >= this.nElements.
216 	 */
217 	buckets.LinkedList.prototype.elementAtIndex = function(index) {
218 
219 		if (index < 0 || index >= this.nElements) {
220 			return undefined;
221 		}
222 		if (index === (this.nElements - 1)) {
223 			return this.lastNode.element;
224 		}
225 
226 		var currentNode = this.firstNode;
227 		for ( var i = 0; i < index; i++) {
228 			currentNode = currentNode.next;
229 		}
230 		return currentNode.element;
231 	};
232 
233 	/**
234 	 * Returns the index in this list of the this.firstNode occurrence of the
235 	 * specified element, or -1 if the List does not contain this element.
236 	 * @this {buckets.LinkedList}
237 	 * @param {Object} element element to search for.
238 	 * @return {number} the index in this list of the this.firstNode occurrence
239 	 * of the specified element, or -1 if the list does not contain this
240 	 * element.
241 	 */
242 	buckets.LinkedList.prototype.indexOf = function(element) {
243 
244 		var currentNode = this.firstNode;
245 		var index = 0;
246 		while (currentNode !== null) {
247 
248 			if (this.equalsF(currentNode.element, element)) {
249 				return index;
250 			}
251 			index++;
252 			currentNode = currentNode.next;
253 		}
254 
255 		return -1;
256 	};
257 
258 	/**
259 	 * Returns true if this list contains the specified element.
260 	 * @this {buckets.LinkedList}
261 	 * @param {Object} element element to search for.
262 	 * @return {boolean} true if this list contains the specified element, false
263 	 * otherwise.
264 	 */
265 	buckets.LinkedList.prototype.contains = function(element) {
266 		return (this.indexOf(element) >= 0);
267 	};
268 
269 	/**
270 	 * Removes the first occurrence of the specified element in this list.
271 	 * @this {buckets.LinkedList}
272 	 * @param {Object} element element to be removed from this list, if present.
273 	 * @return {boolean} true if the list contained the specified element.
274 	 */
275 	buckets.LinkedList.prototype.remove = function(element) {
276 
277 		if (this.nElements < 1) {
278 			return false;
279 		}
280 		else {
281 
282 			var previous = null;
283 			var currentNode = this.firstNode;
284 			while (currentNode !== null) {
285 
286 				if (this.equalsF(currentNode.element, element)) {
287 
288 					if (currentNode === this.firstNode) {
289 						this.firstNode = this.firstNode.next;
290 						if (currentNode === this.lastNode) {
291 							this.lastNode = null;
292 						}
293 						break;
294 					}
295 					else if (currentNode === this.lastNode) {
296 						this.lastNode = previous;
297 					}
298 					previous.next = currentNode.next;
299 					currentNode.next = null;
300 					break;
301 				}
302 				previous = currentNode;
303 				currentNode = currentNode.next;
304 			}
305 		}
306 
307 		this.nElements--;
308 		return true;
309 	};
310 
311 	/**
312 	 * Removes all of the elements from this list.
313 	 * @this {buckets.LinkedList}
314 	 */
315 	buckets.LinkedList.prototype.clear = function() {
316 		this.firstNode = null;
317 		this.lastNode = null;
318 		this.nElements = 0;
319 	};
320 
321 	/**
322 	 * Removes the element at the specified position in this list.
323 	 * @this {buckets.LinkedList}
324 	 * @param {number} index given index.
325 	 * @return {*} removed element or undefined if index < 0 || index >=
326 	 * this.nElements.
327 	 */
328 	buckets.LinkedList.prototype.removeElementAtIndex = function(index) {
329 
330 		if (index < 0 || index >= this.nElements) {
331 			return undefined;
332 		}
333 		var r;
334 		if (this.nElements === 1) {
335 			//First node in the list.
336 			r = this.firstNode.element;
337 			this.firstNode = null;
338 			this.lastNode = null;
339 		}
340 		else {
341 
342 			var previous = null;
343 			var current = this.firstNode;
344 
345 			for ( var i = 0; i < index; i++) {
346 				previous = current;
347 				current = current.next;
348 			}
349 			r = current.element;
350 			if (current === this.lastNode) {
351 				this.lastNode = previous;
352 			}
353 			else if (current === this.firstNode) {
354 				this.firstNode = current.next;
355 			}
356 			if (previous !== null) {
357 				previous.next = current.next;
358 				current.next = null;
359 			}
360 		}
361 		this.nElements--;
362 		return r;
363 	};
364 
365 	/**
366 	 * Returns an iterator of the elements in this list (in proper sequence).<br>
367 	 * <br>
368 	 * The iterator has the following operations:<br>
369 	 * <br>
370 	 * hasNext() Returns true if this list iterator has more elements when
371 	 * traversing the list in the forward direction.<br>
372 	 * next() Returns the next element in the list.<br>
373 	 * remove() Removes from the list the last element that was returned by
374 	 * next.<br>
375 	 * replace() Replaces the this.lastNode element returned by next.
376 	 * @this {buckets.LinkedList}
377 	 * @return {*} an iterator of the elements in this list.
378 	 */
379 	buckets.LinkedList.prototype.iterator = function() {
380 
381 		var previous = null;
382 		var current = null;
383 		var next = this.firstNode;
384 		var it = {};
385 
386 		it.hasNext = function() {
387 			return next !== null;
388 		};
389 		it.next = function() {
390 
391 			if (next === null) {
392 				return undefined;
393 			}
394 			previous = current;
395 			current = next;
396 			next = current.next;
397 			return current.element;
398 		};
399 		it.remove = function() {
400 			var r = current.element;
401 			if (current === this.firstNode) {
402 				if (current === this.lastNode) {
403 					this.firstNode = null;
404 					this.lastNode = null;
405 				}
406 				else {
407 					this.firstNode = next;
408 				}
409 			}
410 			else if (current === this.lastNode) {
411 				this.lastNode = previous;
412 				previous.next = null;
413 			}
414 			else {
415 				previous.next = current.next;
416 			}
417 			this.nElements--;
418 			return r;
419 		};
420 		it.replace = function(elem) {
421 			if (current !== null) {
422 				var temp = current.element;
423 				current.element = elem;
424 				return temp;
425 			}
426 			return undefined;
427 		};
428 		return it;
429 	};

431 	/**
432 	 * Returns an array containing all of the elements in this list in proper
433 	 * sequence.
434 	 * @this {buckets.LinkedList}
435 	 * @return {Array.<*>} an array containing all of the elements in this list
436 	 * in proper sequence.

440 		var currentNode = this.firstNode;
441 		while (currentNode !== null) {
442 			array.push(currentNode.element);
443 			currentNode = currentNode.next;
444 		}
445 		return array;
446 	};

470 	/**
471 	 * Creates an empty dictionary. A dictionary maps keys to values. A map
472 	 * cannot contain duplicate keys; each key can map to at most one value.
473 	 * This dictionary accepts any kind of objects as keys. If the keys are
474 	 * custom objects a function which converts keys to strings must be
475 	 * provided. Example:
476 	 * 
477 	 * <pre>
478 	 * function petToString(pet) {
479 	 * 	return pet.name;
480 	 * }
481 	 * </pre>
482 	 * 
483 	 * @constructor
484 	 * @this {buckets.Dictionary}
485 	 * @param {function(Object):string} toStringFunction optional function used
486 	 * to convert keys to strings. If the keys aren't strings or if toString()
487 	 * is not appropriate, a custom function which receives a key and returns a
488 	 * unique string must be provided.
489 	 */
490 	buckets.Dictionary = function(toStringFunction) {
491 
492 		/**
493 		 * Object holding the key-value pairs.
494 		 * @type {Object}
495 		 * @private
496 		 */
497 		this.table = {};
498 
499 		/**
500 		 * Number of elements in the list.
501 		 * @type {number}
502 		 * @private
503 		 */
504 		this.nElements = 0;
505 
506 		/**
507 		 * Function used to convert keys to strings.
508 		 * @type {function(Object):string}
509 		 * @private
510 		 */
511 		this.toStr = toStringFunction || common.DEFAULT_TO_STRING;
512 	};
513 
514 	/**
515 	 * Returns the value to which this dictionary maps the specified key.
516 	 * Returns undefined if the map contains no mapping for this key.
517 	 * @this {buckets.Dictionary}
518 	 * @param {Object} key key whose associated value is to be returned.
519 	 * @return {*} the value to which this dictionary maps the specified key or
520 	 * undefined if the map contains no mapping for this key.
521 	 */
522 	buckets.Dictionary.prototype.get = function(key) {
523 
524 		var pair = this.table[this.toStr(key)];
525 		if (pair === undefined) {
526 			return undefined;
527 		}
528 		return pair.value;
529 	};
530 
531 	/**
532 	 * Associates the specified value with the specified key in this dictionary.
533 	 * If the dictionary previously contained a mapping for this key, the old
534 	 * value is replaced by the specified value.
535 	 * @this {buckets.Dictionary}
536 	 * @param {Object} key key with which the specified value is to be
537 	 * associated.
538 	 * @param {Object} value value to be associated with the specified key.
539 	 * @return {*} previous value associated with specified key, or undefined if
540 	 * there was no mapping for key.
541 	 */
542 	buckets.Dictionary.prototype.set = function(key, value) {
543 
544 		var ret;
545 		var k = this.toStr(key);
546 		var previousElement = this.table[k];
547 		if (previousElement === undefined) {
548 			this.nElements++;
549 			ret = undefined;
550 		}
551 		else {
552 			ret = previousElement.value;
553 		}
554 		this.table[k] = {
555 			key : key,
556 			value : value
557 		};
558 		return ret;
559 	};
560 
561 	/**
562 	 * Removes the mapping for this key from this dictionary if it is present.
563 	 * @this {buckets.Dictionary}
564 	 * @param {Object} key key whose mapping is to be removed from the
565 	 * dictionary.
566 	 * @return {*} previous value associated with specified key, or undefined if
567 	 * there was no mapping for key.
568 	 */
569 	buckets.Dictionary.prototype.remove = function(key) {
570 		var k = this.toStr(key);
571 		var previousElement = this.table[k];
572 		if (previousElement !== undefined) {
573 			delete this.table[k];
574 			this.nElements--;
575 			return previousElement.value;
576 		}
577 		return undefined;
578 	};
579 
580 	/**
581 	 * Returns an array with the keys contained in this dictionary.
582 	 * @this {buckets.Dictionary}
583 	 * @return {Array} an array with the keys contained in this dictionary.
584 	 */
585 	buckets.Dictionary.prototype.keys = function() {
586 		var array = [];
587 		for ( var name in this.table) {
588 			if (this.table.hasOwnProperty(name)) {
589 				array.push(this.table[name].key);
590 			}
591 		}
592 		return array;
593 	};
594 
595 	/**
596 	 * Returns an array with the values contained in this dictionary.
597 	 * @this {buckets.Dictionary}
598 	 * @return {Array} an array with the values contained in this dictionary.
599 	 */
600 	buckets.Dictionary.prototype.values = function() {
601 		var array = [];
602 		for ( var name in this.table) {
603 			if (this.table.hasOwnProperty(name)) {
604 				array.push(this.table[name].value);
605 			}
606 		}
607 		return array;
608 	};
609 
610 	/**
611 	 * Returns true if this dictionary contains a mapping for the specified key.
612 	 * @this {buckets.Dictionary}
613 	 * @param {Object} key key whose presence in this dictionary is to be
614 	 * tested.
615 	 * @return {boolean} true if this dictionary contains a mapping for the
616 	 * specified key.
617 	 */
618 	buckets.Dictionary.prototype.containsKey = function(key) {
619 		return this.get(key) !== undefined;
620 	};
621 
622 	/**
623 	 * Removes all mappings from this dictionary.
624 	 * @this {buckets.Dictionary}
625 	 */
626 	buckets.Dictionary.prototype.clear = function() {
627 		
628 		this.table = {};
629 		this.nElements = 0;
630 	};
631 
632 	/**
633 	 * Returns the number of key-value mappings in this dictionary.
634 	 * @this {buckets.Dictionary}
635 	 * @return {number} the number of key-value mappings in this dictionary.
636 	 */
637 	buckets.Dictionary.prototype.size = function() {
638 		return this.nElements;
639 	};
640 
641 	/**
642 	 * Returns true if this dictionary contains no mappings.
643 	 * @this {buckets.Dictionary}
644 	 * @return {boolean} true if this dictionary contains no mappings.
645 	 */
646 	buckets.Dictionary.prototype.isEmpty = function() {
647 		return this.nElements <= 0;
648 	};

654 	/**
655 	 * Creates an empty Heap.
656 	 * @class Smaller elements are on top. If the inserted elements are custom
657 	 * objects a compare function must be provided, otherwise the <=, === and >=
658 	 * operators are used to compare objects. Example:
659 	 * 
660 	 * <pre>
661 	 * function comparePetsByAge(pet1, pet2) {
662 	 * 	if (pet1.age < pet2.age) {
663 	 * 		return -1;
664 	 * 	}
665 	 * 	else if (pet1.age > pet2.age) {
666 	 * 		return 1;
667 	 * 	}
668 	 * 	else {
669 	 * 		return 0;
670 	 * 	}
671 	 * }
672 	 * </pre>
673 	 * 
674 	 * If a Max-Heap is wanted (greater elements on top) you can a provide a
675 	 * reverse compare function to accomplish this behavior. Example:
676 	 * 
677 	 * <pre>
678 	 * function reverseCompareNumbers(a, b) {
679 	 * 	if (a < b) {
680 	 * 		return 1;
681 	 * 	}
682 	 * 	else if (a > b) {
683 	 * 		return -1;
684 	 * 	}
685 	 * 	else {
686 	 * 		return 0;
687 	 * 	}
688 	 * }
689 	 * </pre>
690 	 * 
691 	 * @constructor
692 	 * @this {buckets.Heap}
693 	 * @param {function(Object,Object):number=} compareFunction optional
694 	 * function used to compare two elements. Must return a negative integer,
695 	 * zero, or a positive integer as the first argument is less than, equal to,
696 	 * or greater than the second.
697 	 */
698 	buckets.Heap = function(compareFunction) {
699 
700 		/**
701 		 * Array used to store the elements od the heap.
702 		 * @type {Array.<Object>}
703 		 * @private
704 		 */
705 		this.data = [];
706 
707 		/**
708 		 * Function used to compare elements.
709 		 * @type {function(Object,Object):number}
710 		 * @private
711 		 */
712 		this.compare = compareFunction || common.DEFAULT_COMPARE;
713 	};
714 
715 	/**
716 	 * Returns the index of the left child of the node at the given index.
717 	 * @param {number} nodeIndex The index of the node to get the left child
718 	 * for.
719 	 * @return {number} The index of the left child.
720 	 * @private
721 	 */
722 	buckets.Heap.prototype.leftChildIndex = function(nodeIndex) {
723 		return (2 * nodeIndex) + 1;
724 	};
725 
726 	/**
727 	 * Returns the index of the right child of the node at the given index.
728 	 * @param {number} nodeIndex The index of the node to get the right child
729 	 * for.
730 	 * @return {number} The index of the right child.
731 	 * @private
732 	 */
733 	buckets.Heap.prototype.rightChildIndex = function(nodeIndex) {
734 		return (2 * nodeIndex) + 2;
735 	};
736 
737 	/**
738 	 * Returns the index of the parent of the node at the given index.
739 	 * @param {number} nodeIndex The index of the node to get the parent for.
740 	 * @return {number} The index of the parent.
741 	 * @private
742 	 */
743 	buckets.Heap.prototype.parentIndex = function(nodeIndex) {
744 		return Math.floor((nodeIndex - 1) / 2);
745 	};
746 
747 	/**
748 	 * Returns the index of the smaller child node (if it exists).
749 	 * @this {buckets.Heap}
750 	 * @param {number} leftChild left child index.
751 	 * @param {number} rightChild right child index.
752 	 * @return {number} the index with the minimum value or -1 if it doesn't
753 	 * exists.
754 	 * @private
755 	 */
756 	buckets.Heap.prototype.minIndex = function(leftChild, rightChild) {
757 
758 		if (rightChild >= this.data.length) {
759 			if (leftChild >= this.data.length) {
760 				return -1;
761 			}
762 			else {
763 				return leftChild;
764 			}
765 		}
766 		else {
767 			if (this.compare(this.data[leftChild], this.data[rightChild]) <= 0) {
768 				return leftChild;
769 			}
770 			else {
771 				return rightChild;
772 			}
773 		}
774 	};
775 
776 	/**
777 	 * Moves the node at the given index up to its proper place in the heap.
778 	 * @this {buckets.Heap}
779 	 * @param {number} index The index of the node to move up.
780 	 * @private
781 	 */
782 	buckets.Heap.prototype.siftUp = function(index) {
783 
784 		var value = this.data[index];
785 		while (index > 0) {
786 			var parent = this.parentIndex(index);
787 			if (this.compare(this.data[parent], value) > 0) {
788 				this.data[index] = this.data[parent];
789 				index = parent;
790 			}
791 			else {
792 				break;
793 			}
794 		}
795 		this.data[index] = value;
796 	};
797 
798 	/**
799 	 * Moves the node at the given index down to its proper place in the heap.
800 	 * @this {buckets.Heap}
801 	 * @param {number} nodeIndex The index of the node to move down.
802 	 * @private
803 	 */
804 	buckets.Heap.prototype.siftDown = function(nodeIndex) {
805 
806 		var min = this.minIndex(this.leftChildIndex(nodeIndex), this
807 				.rightChildIndex(nodeIndex));
808 
809 		while (min >= 0
810 				&& this.compare(this.data[nodeIndex], this.data[min]) > 0) {
811 			var tmp = this.data[min];
812 			this.data[min] = this.data[nodeIndex];
813 			this.data[nodeIndex] = tmp;
814 			nodeIndex = min;
815 			min = this.minIndex(this.leftChildIndex(nodeIndex), this
816 					.rightChildIndex(nodeIndex));
817 		}
818 	};

820 	/**
821 	 * Retrieves but does not remove the root element of this heap.
822 	 * @this {buckets.Heap}
823 	 * @return {*} The value at the root of the heap. Returns undefined if the
824 	 * heap is empty.
825 	 */
826 	buckets.Heap.prototype.peek = function() {
827 
828 		if (this.data.length > 0) {
829 			return this.data[0];
830 		}
831 		else {
832 			return undefined;
833 		}
834 	};

837 	 * Adds the given element into the heap.
838 	 * @this {buckets.Heap}
839 	 * @param {*} element the element.
840 	 */
841 	buckets.Heap.prototype.add = function(element) {
842 		this.data.push(element);
843 		this.siftUp(this.data.length - 1);
844 	};

846 	/**
847 	 * Retrieves and removes the root element of this heap.
848 	 * @this {buckets.Heap}
849 	 * @return {*} The value removed from the root of the heap. Returns
850 	 * undefined if the heap is empty.
851 	 */
852 	buckets.Heap.prototype.removeRoot = function() {
853 
854 		if (this.data.length > 0) {
855 			var obj = this.data[0];
856 			this.data[0] = this.data[this.data.length - 1];
857 			this.data.splice(this.data.length - 1, 1);
858 			if (this.data.length > 0) {
859 				this.siftDown(0);
860 			}
861 			return obj;
862 		}
863 		return undefined;
864 	};

871 	 * otherwise.

874 
875 		var l = this.data.length;
876 		for ( var i = 0; i < l; i++) {
877 			if (this.compare(this.data[i], element) === 0) {
878 				return true;
879 			}
880 		}
881 		return false;
